374void ManejoPorTeclado()
375{
376 uint8_t tecla_ingresada;
377
378 UartReadByte(UART_PC, &tecla_ingresada); // va a leer la tecla que apreto en el teclado y me va a modificar mi variable
379
380 if (tecla_ingresada == 'T')
381 {
382 PERIODO_MUESTREO_DA = PERIODO_MUESTREO_DA + 100;
383 }
384
385 else if (tecla_ingresada == 'B')
386 {
387 PERIODO_MUESTREO_DA = PERIODO_MUESTREO_DA - 100;
388 }
389
390 else if (tecla_ingresada == 'R')
391 {
392 PERIODO_MUESTREO_DA = 4000;
393 }
394
395 else
396 {
397 printf("Tecla ingresada incorrecta.\n");
398 }
399}

401void app_main(void)
402{
403 SwitchesInit();
404
405 /*Configuro e inicializo el puerto serie*/
406 serial_config_t mi_serial;
407 mi_serial.port = UART_PC;
408 mi_serial.baud_rate = 115200; // tasa de bits por segundo
409 mi_serial.func_p = NULL; // esta funcion va a hacer las interrupciones cuando haya apretado una tecla
410
411 /*Configuro e inicializo la UART*/
412 UartInit(&mi_serial);
413
414 /*Configuro el canal analogico*/
415 analog_input_config_t mi_analogico;
416 mi_analogico.input = CH1;
417 mi_analogico.mode = ADC_SINGLE;
418 mi_analogico.func_p = NULL;
419 mi_analogico.param_p = NULL;
420
421 /*Inicializo la lectura analogica*/
422 AnalogInputInit(&mi_analogico);
423
424 /*Interrupciones para la lectura de los switch*/
425 SwitchActivInt(SWITCH_1, &LeerSwitch, NULL);
426 SwitchActivInt(SWITCH_2, &LeerSwitch, NULL);
427
428 /*Configuro e inicializo el timer para la conversion analogica - digital*/
429 timer_config_t timer_conversion_AD = {
430 .timer = TIMER_A,
431 .period = PERIODO_MUESTREO_AD,
432 .func_p = TimerADC,
433 .param_p = NULL};
434 TimerInit(&timer_conversion_AD);
435
436 /*Configuro e inicializo el timer para la conversion digital - analogica*/
437 timer_config_t timer_conversion_DA = {
438 .timer = TIMER_B,

440 .func_p = TimerDAC,
441 .param_p = NULL};
442 TimerInit(&timer_conversion_DA);
443
444 /*Creo la tarea para la conversion analogica - digital*/
445 xTaskCreate(&ConversionAD, "Conversor AD", 2048, NULL, 5, &conversion_AD_handle);

447 /*Creo la tarea para la conversion digital - analogica*/
