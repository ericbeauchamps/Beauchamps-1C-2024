103void ConversionAD()
104{
105 while (1)
106 {
107 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
108
109 AnalogInputReadSingle(CH1, &valor_medido);
110
111 /*Evita errores a la hora de medir el angulo porque el potenciometro de desplazaba en mi brazo debido a la falta de una buena fijacion*/
112 if (valor_medido < VALOR_MINIMO)
113 {
114 valor_medido = VALOR_MINIMO;
115 }
116
117 /*Evita errores a la hora de medir el angulo porque el potenciometro de desplazaba en mi brazo debido a la falta de una buena fijacion*/
118 if (valor_medido > VALOR_MAXIMO)
119 {
120 valor_medido = VALOR_MAXIMO;
121 }
122
123 /*Realizo un mapeo para hallar el angulo medido*/
124 angulo_medido = ANGULO_MINIMO + (valor_medido - VALOR_MINIMO) * (ANGULO_MAXIMO - ANGULO_MINIMO) / (VALOR_MAXIMO - VALOR_MINIMO);

197 TimerInit(&timer_conversion_AD);

200 /*Tareas necesarias para la resolucion del proyecto*/
201 xTaskCreate(&ConversionAD, "Conversor AD", 2048, NULL, 5, &conversion_AD_handle);
202 xTaskCreate(&Visualizacion, "Visualizacion", 2048, NULL, 5, &visualizacion_handle);
203
204 while (1)
205 {
206 vTaskDelay(CONFIG_BLINK_PERIOD / portTICK_PERIOD_MS);
207
208 /*Manejo del bluetooth*/
209 switch (BleStatus())
210 {
211 case BLE_OFF:
212 LedOff(LED_BT);
213 break;
214 case BLE_DISCONNECTED:
215 LedToggle(LED_BT);
216 break;
217 case BLE_CONNECTED:
218 LedOn(LED_BT);
219
220 /*Bandera booleana que permite realizar las tareas solo cuando el BLE de la placa esta conectado al BLE del dispositivo movil*/
221 bt_conectado = true;

224 }
