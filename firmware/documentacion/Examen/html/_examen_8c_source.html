88void Medicion()
89{
90 while (1)
91 {
92
93 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
94
95 if (irrigacion_habilitada == true) // Controlo si el sistema esta habilitado
96 {
97 GPIOOn(PinSensorHumedad.pin); // Enciendo el GPIO del sensor de humedad
98
99 if (GPIORead(PinSensorHumedad.pin) == true) // Leo el estado actual del sensor y si es 1
100 {
101 estado_bomba_agua = true; // Bandera para saber el estado de la bomba de agua: false signficia apagada y true es prendida
102 estado_humedad = false; // Bandera para saber el estado de humedad: false signficia incorrecta y true es correcta
103 GPIOOn(PinBombas[0].pin); // Prendo la bomba de agua
104 vTaskDelay(CONFIG_BOMBA_PERIODO / portTICK_PERIOD_MS); // delay para dar tiempo para regar durante 1 Seg
105 }
106
107 else
108 {
109 estado_bomba_agua = false; // Bandera para saber el estado de la bomba de agua: false signficia apagada y true es prendida
110 estado_humedad = true; // Bandera para saber el estado de humedad: false signficia incorrecta y true es correcta
111 GPIOOff(PinBombas[0].pin); // Apago la bomba de agua
112 }
113
114 AnalogInputReadSingle(CH1, &valor_pH_milivolts); // leo el valor de pH en milivolts
115
116 /*Regla de 3
117
118 Determino cuando se debe prender la bomba basica
119 3000 mV ------ 14 pH
120 x = 1284.7 mV ------------- 6 pH
121
122 Determino cuando se debe prender la bomba acida
123 3000 mV ------ 14 pH
124 x = 1435.7 mV ------------- 6,7 pH*/
125
126 valor_pH_real = (valor_pH_milivolts*14)/VOLTAJE_REFERENCIA; //acá el pH real ya está listo por regla de 3
127
128 if (valor_pH_milivolts < VALOR_BOMBA_BASICA)
129 {
130
131 GPIOOn(PinBombas[2].pin); // Prendo la bomba basica
132 estado_bomba_basica = true;
133 vTaskDelay(CONFIG_BOMBA_PERIODO / portTICK_PERIOD_MS); // delay para dar tiempo para regar durante 1 Seg
134 GPIOOff(PinBombas[2].pin); // Apago la bomba basica
135 estado_bomba_basica = false;
136 }
137
138 else if (valor_pH_milivolts > VALOR_BOMBA_ACIDA)
139 {
140 GPIOOn(PinBombas[1].pin); // Prendo la bomba acida
141 estado_bomba_acida = true;
142 vTaskDelay(CONFIG_BOMBA_PERIODO / portTICK_PERIOD_MS); // delay para dar tiempo para regar durante 1 Seg
143 GPIOOff(PinBombas[1].pin); // Prendo la bomba acida
144 estado_bomba_acida = false;
145 }
146 }
147
148 else
149 {
150 GPIOOff(PinSensorHumedad.pin); // Apago los GPIO
151 GPIOOff(PinBombas[0].pin); // Apago los GPIO
152 GPIOOff(PinBombas[1].pin); // Apago los GPIO
153 GPIOOff(PinBombas[2].pin); // Apago los GPIO
154 }
155 }
156}

236void app_main(void)
237{

245
246 /*Configuro e GPIO para las bombas*/

248 PinBombas[0].dir = GPIO_OUTPUT;
249 PinBombas[1].pin = GPIO_21; // Esta es la bomba de pHA
250 PinBombas[1].dir = GPIO_OUTPUT;
251 PinBombas[2].pin = GPIO_22; // Esta es la bomba de pHB
252 PinBombas[2].dir = GPIO_OUTPUT;
253 GPIOInit(PinBombas[0].pin, PinBombas[0].dir);
254 GPIOInit(PinBombas[1].pin, PinBombas[1].dir);
255 GPIOInit(PinBombas[2].pin, PinBombas[2].dir);
256
257 /*Configuro e inicializo el puerto serie*/
258 serial_config_t mi_serial;
259 mi_serial.port = UART_PC;
260 mi_serial.baud_rate = 115200;
261 mi_serial.func_p = NULL;
262 UartInit(&mi_serial);
263
264 /*Configuro e inicializo el timer para la medicion*/
265 timer_config_t timer_medicion = {
266 .timer = TIMER_A,
267 .period = PERIODO_MEDICION,
268 .func_p = TimerMedicion,
269 .param_p = NULL};
270 TimerInit(&timer_medicion);
271
272 xTaskCreate(&Medicion, "Medicion", 2048, NULL, 5, &medicion_handle);
273
274 /*Configuro e inicializo el timer para informar el estado*/
275 timer_config_t timer_estado = {
276 .timer = TIMER_B,
277 .period = PERIODO_ESTADO,
278 .func_p = TimerEstado,
279 .param_p = NULL};
280 TimerInit(&timer_estado);
281
282 xTaskCreate(&Estado, "Estado", 2048, NULL, 5, &estado_handle);
283
284 while (1)
285 {
286 /*Hago la lectura de las teclas para habilitar/desabilitar el sistema*/
287 teclas = SwitchesRead();
288 switch (teclas)

290 case SWITCH_1:
291 irrigacion_habilitada = true;
292 break;
293 case SWITCH_2:
294 irrigacion_habilitada = false;
295 break;
296 }

298 vTaskDelay(CONFIG_SWITCH_PERIODO / portTICK_PERIOD_MS);
299 }
300
301 TimerStart(TIMER_A);
302 TimerStart(TIMER_B);
303}
