152void MostrarDistancia(void *pvParameter)
153{
154 while (1)
155 {
156 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
157
158 if (control_medicion && !control_hold)
159 {
160 LcdItsE0803Write(distancia_medida);
161
162 // UartSendString: manda un string al puerto que paso
163 /* UartItoa: convierte un entero sin signo de 32 bits a una
164 cadena de caracteres en una base especificada y devuelve un puntero a
165 la cadena resultante. La cadena resultante se almacena en un búfer
166 estático buf de tamaño 32.*/
167
168 if(control_unidad)
169 {
170 UartSendString(UART_PC,"Distancia medida: ");
171 UartSendString(UART_PC, (const char*)UartItoa(distancia_medida, 10));
172 UartSendString(UART_PC," ");
173 UartSendString(UART_PC,"cm");
174 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
175 }
176
177 else if(!control_unidad)
178 {
179 UartSendString(UART_PC,"Distancia medida: ");
180 UartSendString(UART_PC, (const char*)UartItoa(distancia_medida*constante_pulgadas, 10));
181 UartSendString(UART_PC," ");
182 UartSendString(UART_PC,"pulgadas");
183 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
184 }
185 }
186
187 else if (!control_medicion)
188 {
189 LcdItsE0803Off();
190 }
191 // vTaskDelay(MEDICION_PERIODO / portTICK_PERIOD_MS);
192 }
193}

246void ManejoPorTeclado()
247{
248 uint8_t tecla_ingresada;
249
250 UartReadByte(UART_PC, &tecla_ingresada);//va a leer la tecla que apreto en el teclado y me va a modificar mi variable
251
252 if (tecla_ingresada == 'O')
253 {
254 FuncionTecla1();
255 }
256
257 else if (tecla_ingresada == 'H')
258 {
259 FuncionTecla2();
260 }
261
262 else if (tecla_ingresada == 'I')
263 {
264 control_unidad = !control_unidad;
265 }
266
267 else if (tecla_ingresada == 'M')
268 {
269 UartSendString(UART_PC,"Distancia maxima: ");
270 UartSendString(UART_PC, (const char*)UartItoa(distancia_auxiliar, 10));
271 UartSendString(UART_PC," ");
272 UartSendString(UART_PC,"cm");
273 UartSendString(UART_PC,"\r\n"); //esto es el fin de linea abajo
274 }
275
276 else
277 {
278 printf("Tecla ingresada incorrecta.\n");
279 }
280}

285 /*Actividad 3 - Proyecto: Medidor de distancia por ultrasonido c/interrupciones y puerto serie
286
287 Modifique la actividad del punto 2 agregando ahora el puerto serie.
288 Envíe los datos de las mediciones para poder observarlos en un terminal en la PC,
289 siguiendo el siguiente formato:
290
291 3 dígitos ascii + 1 carácter espacio + dos caracteres para la unidad (cm) + cambio de línea “ \r\n”
292 Además debe ser posible controlar la EDU-ESP de la siguiente manera:
293 Con las teclas “O” y “H”, replicar la funcionalidad de las teclas 1 y 2 de la EDU-ESP
294 */
295
296 LedsInit();
297 HcSr04Init(GPIO_3, GPIO_2);
298 LcdItsE0803Init();
299 SwitchesInit();
300
301 /*Configuro e inicializo el puerto serie*/
302 serial_config_t mi_serial;
303 mi_serial.port = UART_PC;
304 mi_serial.baud_rate = 115200; // tasa de bits por segundo
305 mi_serial.func_p = ManejoPorTeclado; //esta funcion va a hacer las interrupciones cuando haya apretado una tecla
306
307 /*Configuro e inicializo la UART*/
308 UartInit(&mi_serial);
309
310 /*inicializo el timer*/
311 timer_config_t timer_distancia = {
312 .timer = TIMER_A,
313 .period = MEDICION_PERIODO,
314 .func_p = FuncTimerA,
315 .param_p = NULL};
316 TimerInit(&timer_distancia);
317
318 SwitchActivInt(SWITCH_1, &LeerSwitch, NULL); //interrupción
319 SwitchActivInt(SWITCH_2, &LeerSwitch, NULL);
320
321 xTaskCreate(&MedirDistancia, "Medir distancia", 2048, NULL, 5, &medir_distancia_handle); //tarea
322 xTaskCreate(&MostrarDistancia, "Mostrar distancia", 2048, NULL, 5, &mostrar_distancia_handle);
323
324 TimerStart(timer_distancia.timer);
